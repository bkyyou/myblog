# 3.1-事件处理

# React 事件和 DOM 事件

  • 第一种方式
class App extends Component {
  handleAdd(ev) {
    // ev 事件对象
  }
  render() {
    return (
      <div onClick={this.handleAdd}></div>
    )
  }
}

# this 关键字

  1. 在 JSX 中使用 bind 方法
  2. 在构造函数中使用 bind 方法
  3. 使用箭头函数
  • 第二种方式
class App extends Component {
  handleAdd(ev) {
    // ev 事件对象
  }
  render() {
    return (
      <div onClick={this.handleAdd.bind(this)}></div>
    )
  }
}
  • 第三种方式
class App extends Component {
  constructor(props) {
    super(props)

    this.handleAdd = this.handleAdd.bind(this)
  }
  handleAdd(ev) {
    // ev 事件对象
  }
  render() {
    return (
      <div onClick={this.handleAdd}></div>
    )
  }
}
  • 第四种方式
class App extends Component {
  constructor(props) {
    super(props)
  }
  handleAdd = (ev) => {
    // ev 事件对象
  }
  render() {
    return (
      <div onClick={this.handleAdd}></div>
    )
  }
}

# 向事件处理程序传递参数

  • 第一种方式
class App extends Component {
  constructor(props) {
    super(props)
    this.state = {

    }
  }
  // handleAdd(ev) {
  //   // ev 事件对象
  // }
  // handleAdd(id) {
  //   // ev 事件对象
  // }
  handleAdd = (id) => {
    // ev 事件对象
  }
  render() {
    return (
      <div onClick={this.handleAdd.bind(this, this.props.data.id)}></div>
    )
  }
}
  • 第二种方式
class App extends Component {
  constructor(props) {
    super(props)
    this.state = {
      id: 1
    }
  }
  handleAdd = (id) => {
    // ev 事件对象
  }
  doHandle = () => {
    // ev 事件对象
    this.handleAdd(this.state.id)
  }
  render() {
    return (
      <div onClick={this.doHandle}></div>
    )
  }
}
  • 第三种方式
class App extends Component {
  constructor(props) {
    super(props)
    this.state = {
      id: 1
    }
  }
  handleAdd = (id, ev) => {
    // ev 事件对象
  }
  render() {
    return (
      <div onClick={(ev) => { this.handleAdd(this.props.data.id, ev) }}></div>
    )
  }
}
