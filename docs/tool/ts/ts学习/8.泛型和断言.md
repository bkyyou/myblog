# 8.泛型和断言

如果我们编写了一个公用的工具，我们需要接收不同的参数来做不同的处理，怎么处理，也就是说我们不明确他需要什么类型，这个时候可以用泛型

## 8.1.泛型

### 8.1.1泛型编写

```ts
// 泛型编写
function fn2<T>(a: T): T {
  return a
}
```

```ts
// 实战
// find - 数组， 目标， 找出个目标
// [1, '2', 3] 3
function find<T>(arg: T[], target: T): T {
  let resulet;

  for (const val of arg) {
    if (val === target) {
      resulet = val;
    }
  }

  return resulet;
}

// 可以设置默认值
function find<T = number>(arg: T[], target: T): T {
  let resulet;

  for (const val of arg) {
    if (val === target) {
      resulet = val;
    }
  }

  return resulet;
}

find([1,'2'], 1)

// 可以调用的时候指定类型
find<string>(['1','2'], '1');
```

```ts
// 泛型有两个
function objExtend<T, U>(a: T, b: U): (T & U) {
  return Object.assign(a, b);
```

```ts
  a1: T,
```

### 8.1.2泛型约束

```ts
// 泛型约束
function find1<T extends number | string>(arg: T[], target: T): T {
  let resulet;

  for (const val of arg) {
    if (val === target) {
      resulet = val;
    }
  }

  return resulet;
}
```

## 断言

JavaScript 是灵活的，而 ts 是严格类型的，这导致他们可能会存在冲突。我们可以用 as 断言来解决这个冲突

```ts
interface Obj {
  a: number,
  b: number
}
const obj1 = {} as Obj
obj1.a = 1;

// 不给 as 会报错
const obj1: Obj = {} as Obj

function fn4(a: number[] | string) {
  // 这样 foreach 不会报错
  const _a = a as number[];
  _a.forEach(val => val)
}
```

```ts
interface fn5 {
  (a: number): string,
  a: number
}
const fn6: fn5 = function(a: number) {
  return '111'
} as fn5
fn6.a = 1;

// 另一种写法断言 不用 as 
interface fn5 {
  (a: number): string,
  a: number
}
const fn6: fn5 = <fn5>function(a: number) {
  return '111'
}
fn6.a = 1;
```
